export async function validateDependencies(packageJsonString: string): Promise<{ passed: boolean; errors: string[] }> {
  const errors: string[] = [];
  
  let pkg;
  try {
    pkg = JSON.parse(packageJsonString);
  } catch (e) {
    return { passed: false, errors: ["Golden Path #10: Invalid package.json format generated."] };
  }

  const deps = { ...pkg.dependencies, ...pkg.devDependencies };
  if (Object.keys(deps).length === 0) return { passed: true, errors: [] };

  // STRICT REQUIREMENT: Execute network calls concurrently to preserve engine velocity.
  await Promise.all(Object.entries(deps).map(async ([name, version]) => {
    const cleanVersion = (version as string).replace(/[\^~><=]/g, '').trim();

    try {
      // 1. Slopsquatting & Hallucination Guard
      const npmRes = await fetch(`https://registry.npmjs.org/${name}`);
      
      if (npmRes.status === 404) {
        errors.push(`[Hallucination] Package '${name}' does not exist on npm.`);
        return; 
      }

      const npmData = await npmRes.json();
      const createdStr = npmData.time?.created;
      
      if (createdStr) {
        const ageInHours = (Date.now() - new Date(createdStr).getTime()) / (1000 * 60 * 60);
        if (ageInHours < 48) {
          errors.push(`[Slopsquatting] Package '${name}' is suspiciously new (< 48 hours old).`);
        }
      }

      // 2. Vulnerability Guard (OSV API)
      const osvRes = await fetch("https://api.osv.dev/v1/query", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          version: cleanVersion,
          package: { name: name, ecosystem: "npm" }
        })
      });

      if (osvRes.ok) {
        const osvData = await osvRes.json();
        if (osvData.vulns && osvData.vulns.length > 0) {
          errors.push(`[CVE] Package '${name}@${cleanVersion}' has known vulnerabilities in the OSV database.`);
        }
      }
    } catch (err) {
      console.error(`[Dependency Audit] Failed to validate package ${name}:`, err);
      // Failsafe: Log network errors but do not crash the engine loop.
    }
  }));

  return {
    passed: errors.length === 0,
    errors
  };
}
